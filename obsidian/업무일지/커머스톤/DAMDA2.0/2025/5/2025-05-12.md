#!/bin/bash

LOG_DIR_LIST=(
    "/app/portal/oauth/logs"
    "/app/portal/tomcat/logs"
    "/app/openAPI/OpenAPI/analytics/wso2/worker/logs"
    "/app/openAPI/OpenAPI/apim/2.6.0/repository/logs"
    "/app/openAPI/OpenAPI/ei/wso2/business-process/repository/logs"
    "/app/portal/logs"
)

TODAY=$(date +%Y%m%d)
DAYS=90

for BASE_DIR in "${LOG_DIR_LIST[@]}"; do
    echo "▶ 처리 중: $BASE_DIR"

    BACKUP_DIR="$BASE_DIR/log_backup"
    ARCHIVE_NAME="${BACKUP_DIR}/${TODAY}_logs_backup.tar.gz"
    mkdir -p "$BACKUP_DIR"

    find "$BASE_DIR" -maxdepth 1 -type f \( \
        -name "*.log" -o \
        -name "*.log.*" -o \
        -name "*.log.*.gz" -o \
        -name "*.log.*.txt" -o \
        -name "*.log-*" -o \
        -name "*.txt" \
    \) -mtime +$DAYS ! -path "${BACKUP_DIR}/*" -exec mv -t "$BACKUP_DIR" {} +

    cd "$BACKUP_DIR" || continue
    if ls * 1> /dev/null 2>&1; then
        tar -czf "$ARCHIVE_NAME" *
        find "$BACKUP_DIR" -maxdepth 1 -type f ! -name "$(basename "$ARCHIVE_NAME")" -exec rm -f {} +
        echo "[$BASE_DIR] → $ARCHIVE_NAME 백업 완료"
    else
        echo "[$BASE_DIR] 백업 대상 로그 없음"
    fi

    for SUBDIR in "$BASE_DIR"/*; do
        [ -d "$SUBDIR" ] || continue
        DIR_NAME=$(basename "$SUBDIR")

        [[ "$DIR_NAME" == "archive" || "$DIR_NAME" == "log_backup" ]] && continue

        BACKUP_DIR="$SUBDIR/log_backup"
        ARCHIVE_NAME="${BACKUP_DIR}/${TODAY}_logs_backup.tar.gz"
        mkdir -p "$BACKUP_DIR"

        find "$SUBDIR" -type f \( \
            -name "*.log" -o \
            -name "*.log.*" -o \
            -name "*.log.*.gz" -o \
            -name "*.log.*.txt" -o \
            -name "*.log-*" -o \
            -name "*.txt" \
        \) -mtime +$DAYS ! -path "${BACKUP_DIR}/*" -exec mv -t "$BACKUP_DIR" {} +

        cd "$BACKUP_DIR" || continue
        if ls * 1> /dev/null 2>&1; then
            tar -czf "$ARCHIVE_NAME" *
            find "$BACKUP_DIR" -maxdepth 1 -type f ! -name "$(basename "$ARCHIVE_NAME")" -exec rm -f {} +
            echo "[$SUBDIR] → $ARCHIVE_NAME 백업 완료"
        else
            echo "[$SUBDIR] 백업 대상 로그 없음"
        fi
    done
done
